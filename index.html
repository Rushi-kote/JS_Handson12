Q3: Add a method to calculate sum of all elements in Array in array's protype, use that method to calculate sum for multiple arrays

Ans:
    const ArraySum = {
        Sum: function (arr) {
        let sum = 0;
        for (let i = 0; i < arr.length; i++) {
            sum += arr[i];
        }
        console.log('sum: ', sum);
        }
    };
    
    ArraySum.Sum([1,2,3,4,5,6,7,8,9]);
